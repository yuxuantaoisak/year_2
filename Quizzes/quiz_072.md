def data_sorting(data):
    capitals = []
    countries = []
    for i in range(0, len(data), 2):
        capitals.append(data[i])
        countries.append(data[i+1])
    return capitals, countries


europe = ["Ankara", "Turkey", "Tokyo", "Japan", "Lisbon", "Portugal"]
print(data_sorting(europe))
